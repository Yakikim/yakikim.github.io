SELECT user_name  FROM

(SELECT DISTINCT  FU.USER_NAME,  aida.creation_date

  FROM PO_HEADERS_ALL       PHA,

       PO_DISTRIBUTIONS_ALL PDA,

       FND_USER FU,

       AP_INVOICE_DISTRIBUTIONS_ALL AIDA,

       AP_INVOICES_ALL      AIA

 WHERE PHA.PO_HEADER_ID = PDA.PO_HEADER_ID

   AND PDA.PO_DISTRIBUTION_ID = AIDA.PO_DISTRIBUTION_ID

   AND AIDA.INVOICE_ID = AIA.INVOICE_ID

   AND pda.deliver_to_person_id = FU.EMPLOYEE_ID

   AND SYSDATE BETWEEN FU.START_DATE AND NVL(FU.END_DATE, SYSDATE +1)

   AND PHA.TYPE_LOOKUP_CODE = 'STANDARD'

   AND AIA.INVOICE_ID = <P_INVOICE_ID>

SELECT DISTINCT FU.USER_NAME,  aida.creation_date

  FROM PO_RELEASES_ALL PRA ,

       PO_DISTRIBUTIONS_ALL PDA,

       FND_USER FU,

       AP_INVOICE_0_ALL AIDA,

       AP_INVOICES_ALL      AIA

 WHERE PRA.PO_RELEASE_ID = PDA.PO_RELEASE_ID

   AND PDA.PO_DISTRIBUTION_ID = AIDA.PO_DISTRIBUTION_ID

   AND AIDA.INVOICE_ID = AIA.INVOICE_ID

FUNCTION ADD_RESPONSE (

        P_SURVEY_ID            NUMBER,

        P_EMAIL_ADDRESS      VARCHAR2)   RETURN VARCHAR2

     IS

        I               NUMBER;

        L_Q_KEY      VARCHAR2(16);

    BEGIN

        I := 1;

        WHILE I <= 1 LOOP

            BEGIN

                L_Q_KEY := DBMS_RANDOM.STRING('U', 16);

                INSERT INTO EBA_SB_RESPONSES (ID, SURVEY_ID, EMAIL_ADDR) VALUES (L_Q_KEY, P_SURVEY_ID, P_EMAIL_ADDRESS);

                I := I + 1;

            EXCEPTION

                WHEN DUP_VAL_ON_INDEX THEN

                    NULL; -- IGNORE THE ERROR AND TRY AGAIN

            END;

        END LOOP ;

        COMMIT;

        EBA_SB.SURVEY_JOB(P_APP_ID => 141,

                                     P_EMAIL_ONLY => TRUE);                   

        RETURN L_Q_KEY;

    END ADD_RESPONSE;
